\section*{DIAG - Diagnose Instruction}
\addcontentsline{toc}{section}{DIAG - Diagnose Instruction}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{DIAG info1, RegB, info2, RegA}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{DIAG} instruction uses the indexed instruction format, identical to the TRAP instruction. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,9.5,11.5,16}
    {\OpCodeGrpSYS, \OpCodeDIAG, RegR, info1, RegB, info2, RegA, 0}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{DIAG} instruction provides a mechanism for diagnostic and privileged operations.  

The \texttt{info1} and \texttt{info2} fields specify the diagnostic function. The \texttt{RegB} and \texttt{RegA} fields may provide operands or data required by the diagnostic function. Any result is returned in \texttt{RegR}.
 
\end{iPageDescEntry}
