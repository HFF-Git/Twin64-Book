\pagebreak
\section*{LPA — Load Physical Address}
\addcontentsline{toc}{section}{LPA — Load Physical Address}

\begin{iPageDescEntry}{Syntax}
    \texttt{LPA RegR, (RegB, RegX)}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{LPA} instruction uses the indexed load instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0,1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodeLPA, RegR, 0, RegB, 0, RegX, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{LPA} (Load Physical Address) instruction computes the physical 
    address corresponding to the given virtual address and loads it into 
    \texttt{RegR}.  
    The virtual address is formed from the contents of \texttt{RegB} and 
    (optionally) \texttt{RegX}, using the same addressing modes as normal loads.  
    No memory access is performed.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{RegR ← translateVA(RegB + RegX);}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    - Page fault if the virtual address is not mapped. \\
    - Privileged instruction trap if executed in user mode.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    - Useful for operating systems to probe or manage virtual→physical mappings. \\
    - No data is loaded from memory; only the translation is returned. \\
    - May be used with TLB management instructions such as \texttt{ITLB}.
\end{iPageDescEntry}
