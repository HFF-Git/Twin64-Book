\section*{SSM — Set System Mask}
\addcontentsline{toc}{section}{SSM — Set System Mask}

\begin{iPageDescEntry}{Syntax}
    \texttt{SSM RegR, mask}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{SSM} instruction uses the system instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0,1,3,5,13,13.5,14,14.5,15,15.5,16}
        {\OpCodeGrpSYS, \OpCodeSSM, RegR, 0, 0, 0, 0, 0, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{SSM} (Set System Mask) instruction updates the system mask 
    register from the value in \texttt{RegR}.  
    The system mask typically controls which classes of interrupts or traps 
    are enabled or disabled.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{SystemMask ← RegR;}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    - Privileged instruction trap if executed in user mode.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    - Used by the operating system to control interrupt levels. \\
    - Often paired with \texttt{RFI} or similar return-from-trap sequences. \\
    - Some implementations may mask specific interrupt classes instead of a flat bitmask.
\end{iPageDescEntry}
