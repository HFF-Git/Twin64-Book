\pagebreak
\section*{MFCR — Move From Control Register}
\addcontentsline{toc}{section}{MFCR — Move From Control Register}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{MFCR RegR, CRegB}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{MFCR} instruction uses the control-register instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,16}
    {\OpCodeGrpSYS, \OpCodeMFCR, RegR, 0, CReg, 0}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{MFCR} instruction copies the contents of control register \texttt{CRegB} to general register \texttt{RegR}.  
Some control registers are privileged; attempting to read them in user mode may cause a privilege exception.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- CRegN;
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
Privilege exception if a privileged control register is accessed in user mode.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None
\end{iPageDescEntry}
