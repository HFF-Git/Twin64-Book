\pagebreak
\section*{MFCR — Move From Control Register}
\addcontentsline{toc}{section}{MFCR — Move From Control Register}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{MFCR RegR, CReg}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{MFCR} instruction uses the control-register instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,13,16}
    {\OpCodeGrpSYS, \OpCodeMFCR, RegR, 0, 0, 0, CReg}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{MFCR} instruction copies the contents of control register \texttt{CReg} to general register \texttt{RegR}. Some control registers are privileged. Attempting to read them in user mode results in a privilege exception.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- CReg;
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
- Privileged Operation Trap for some control registers.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None
\end{iPageDescEntry}
