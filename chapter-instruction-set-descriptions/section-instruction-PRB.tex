\section*{PRB — Probe Virtual Address}
\addcontentsline{toc}{section}{PRB — Probe Virtual Address}

\begin{iPageDescEntry}{Syntax}
    \texttt{PRB RegR, RegB, RegX}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{PRB} instruction uses the register and special-9 instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0, 1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePRB, RegR, 0, RegB, 0, RegA, 0}
    \end{flushleft}
    
    \begin{flushleft}
    \drawInstrFormat
        {0, 1,3,5,6.5,8.5,10.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePRB, RegR, 1, RegB, 0, mode, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{PRB} instruction probes the translation lookaside buffer (TLB) 
    to check whether the effective address in \texttt{RegB} has a valid mapping. The result of the probe is returned in \texttt{RegR}. For the register format instruction \texttt{RegA} contains the mode. For the special-9 format, the mode is encoded in the instruction field mode.

    If the address is not mapped, an appropriate fault code may be placed in \texttt{RegR}.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{RegR ← probeVirtualAdress( RegB, RegA.[62..63]);} \\
    \texttt{RegR ← probeVirtualAdress( RegB, mode);}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    - Memory reference trap (if effective address invalid) \\
    - Privilege violation (if executed in user mode)
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    Typically used by the operating system for virtual memory management.  
\end{iPageDescEntry}
