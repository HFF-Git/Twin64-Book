\pagebreak
\section*{SHRxA - Shift Right and Add}
\addcontentsline{toc}{section}{SHRxA - Shift Right and Add}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{SHRxA RegR, RegB, RegA} \\
\texttt{SHRxA RegR, RegB, val}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{SHRxA} instruction uses the following instruction formats: \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5, 9.5,11.5,16}
    {0,\OpCodeSHRxA,RegR,2,RegB,amt,RegA,0}
\end{flushleft}

\medskip
\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5, 9.5, 16}
    {0,\OpCodeSHRxA,RegR,3,RegB,amt,val}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{SHRxA} instruction shifts the contents of \texttt{RegB} right by the amount specified in the \texttt{amt} field. In register format, \texttt{RegA} is added to the shifted value. In immediate format the sign-extended \texttt{val} field is added instead. The resulting value is stored in \texttt{RegR}.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- ( RegB >> instr.[amt] ) + RegA;                 (Register )
RegR <- ( RegB >> instr.[amt] ) + immOperand( instr );  (Immediate)
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
Integer Overflow Trap
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None
\end{iPageDescEntry}
