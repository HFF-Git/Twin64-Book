\section*{PTLB — Purge TLB Entry}
\addcontentsline{toc}{section}{PTLB — Purge TLB Entry}

\begin{iPageDescEntry}{Syntax}
    \texttt{PTLB RegR, RegB, RegA}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{PTLB} instruction uses the indexed instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0, 1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePTLB, RegR, 1, RegB, 0, RegA, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{PTLB} instruction purges (removes) an entry from the translation 
    lookaside buffer (TLB). The entry to remove is identified by the effective 
    address formed from \texttt{RegR} and \texttt{RegB}, and additional control 
    information from \texttt{RegA}.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{RegR <- purgeFromTlb( RegB, RegA );}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    - Memory reference trap (if address invalid) \\
    - Privilege violation (if executed in user mode)
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    Only available in supervisor mode.  
\end{iPageDescEntry}
