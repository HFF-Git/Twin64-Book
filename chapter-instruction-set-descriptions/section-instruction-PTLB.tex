\section*{PxTLB — Purge TLB Entry}
\addcontentsline{toc}{section}{PxTLB — Purge TLB Entry}

\begin{iPageDescEntry}{Syntax}
    \texttt{PITLB RegR, RegB, RegA} \\
    \texttt{PDTLB RegR, RegB, RegA}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{PTLB} instruction uses the indexed instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
       {0,1,3,5,5.5,6,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePTLB, RegR, 0, M, 1, RegB, 0, RegX, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{PITLB} and \texttt{PDTLB} instructions purge (removes) an entry from the translation lookaside buffer (TLB). The entry to remove is identified by the effective address formed from \texttt{RegB} and \texttt{RegX}. The mode field indicates the TLB type. A value of 0 refers to the instruction TLB, a value of one refers to the data TLB. A value of two refers to a unified TLB. This is privileged instruction.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{RegR <- purgeFromTlb( RegB, RegA );}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    - Privileged operation trap.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
   None.
\end{iPageDescEntry}
