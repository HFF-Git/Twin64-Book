\section*{PxTLB — Purge TLB Entry}
\addcontentsline{toc}{section}{PxTLB — Purge TLB Entry}

\begin{iPageDescEntry}{Syntax}
    \texttt{PITLB RegR, RegB, RegA} \\
    \texttt{PDTLB RegR, RegB, RegA}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{PITLB} instruction uses the following instruction format: \\

    \begin{flushleft}
    \drawInstrFormat
       {0,1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePTLB, RegR, 2, RegB, 0, RegX, 0}
    \end{flushleft}
    
    The \texttt{PDTLB} instruction uses the following instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
       {0,1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePTLB, RegR, 3, RegB, 0, RegX, 0}
    \end{flushleft}
    
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{PxTLB} instructions purge (remove) an entry from the translation lookaside buffer (TLB). The entry to remove is identified by the effective address formed from \texttt{RegB} and \texttt{RegX}. If the M bit is set, the base register RegB is updated with the computed effective address.This is privileged instruction.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{RegR <- purgeFromTlb( addOfs32( RegB, RegX ));}
\end{iPageDescEntry}


\begin{iPageDescEntry}{Exceptions}
    - Privileged operation trap.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
   None.
\end{iPageDescEntry}
