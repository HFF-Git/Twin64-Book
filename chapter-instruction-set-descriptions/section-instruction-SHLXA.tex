\instructionentry{Operation}{


  	\texttt{RegR <-(  RegB << amt ) + RegA} {(register format)}\\
  	\smallskip
  	\texttt{RegR <- ( RegB << amt ) + immOperand( )} {(immediate format)}\\
}

}
