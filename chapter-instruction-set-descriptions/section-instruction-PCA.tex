\section*{PCA - Purge from Cache}
\addcontentsline{toc}{section}{PCA - Purge from Cache}

\begin{iPageDescEntry}{Syntax}
    \texttt{PCA RegR, RegB}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{PCA} instruction uses the standard indexed instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0,1,3,5,6.5,8.5,16}
        {\OpCodeGrpSYS, \OpCodePCA, RegR, 0, RegB, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{PCA} instruction purges a cache line, selected by the effective address
    in \texttt{RegB}. This can be used by privileged code for cache management. Unlike \texttt{FCA}, purge may also invalidate dirty lines without writing them back.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{ RegR <- purgeFromCache( RegB );}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    None.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    Typically only allowed in supervisor mode.
\end{iPageDescEntry}
