\section*{PxCA - Purge from Cache}
\addcontentsline{toc}{section}{PCA - Purge from Cache}

\begin{iPageDescEntry}{Syntax}
    \texttt{PICA RegR, RegB} \\
    \texttt{PDCA RegR, RegB}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{PCA} instruction uses the standard indexed instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0,1,3,5,5.5,6,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePCA, RegR, 0, M, 0, RegB, mode, RegX, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{PICA} and \testtt{PDCA} instructions purge a cache line, selected by the effective address
    in \texttt{RegB}. This can be used by privileged code for cache management. Unlike \texttt{FDCA}, purge may also invalidate dirty lines without writing them back. The mode field indicates the cache type. A value of 0 refers to the instruction cache, a value of one refers to the data cache. A value of two refers to a unified cache.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{ RegR <- purgeFromCache( addOfs32( RegB, RegX );}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    None.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    Typically only allowed in supervisor mode.
\end{iPageDescEntry}
