\section*{ABR - Add and branch}
\addcontentsline{toc}{section}{ABR - Add and branch}

%--------------------------------------------------------------------------------
\instructionentry{Syntax}{

	\texttt{ABR.<cond> RegR, RegB, target}
    
   	where cond is EQ, LT, NE or LE.
}

%--------------------------------------------------------------------------------
\instructionentry{Format}{

	The ABR instruction uses the following format:
	
	\begin{flushleft}
	\drawInstrFormat
  	{0,1,3,5,6.5,8.5,16}
  	{\OpCodeGrpBR,\OpCodeABR,Reg R,Opt 1,Reg B,S-IMM-15/special}
  	\end{flushleft}

}

%--------------------------------------------------------------------------------
\instructionentry{Description}{

The \texttt{ABR} instruction ...

}

%--------------------------------------------------------------------------------
\instructionentry{Operation}{

	\begin{flushleft}
	\hspace*{0em}$\texttt{RegR <- RegR + RegB; }$ \\
	\hspace*{0em}$\texttt{if ( cond( RegR )) IA <- Ia + ofs; }$ \\
	\hspace*{0em}$\texttt{else IA <- Ia + 4; }$
	\end{flushleft}
} 

%--------------------------------------------------------------------------------
\instructionentry{Exceptions}{

	None.
}

%--------------------------------------------------------------------------------
\instructionentry{Notes}{

	None. 
}
