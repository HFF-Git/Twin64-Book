\section*{ITLB — Insert TLB Entry}
\addcontentsline{toc}{section}{ITLB — Insert TLB Entry}

\begin{iPageDescEntry}{Syntax}
    \texttt{ITLB RegR, RegB, RegX \dots}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{ITLB} instruction uses the indexed instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0, 1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePTLB, RegR, 0, RegB, 0, RegA, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{ITLB} instruction inserts a new entry into the translation lookaside
    buffer (TLB). The effective address is formed from \texttt{RegR} and
    \texttt{RegB}, and additional control information is provided by \texttt{RegX}.
    
    Depending on the addressing mode, the instruction may also cause a memory reference trap.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{RegR <- insertIntoTlb( RegB, RegA );}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    - Memory reference trap (if address invalid) \\
    - Privilege violation (if executed in user mode)
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    Only available in supervisor mode.  
\end{iPageDescEntry}
