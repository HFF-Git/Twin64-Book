\pagebreak
\section*{IxTLB — Insert TLB Entry}
\addcontentsline{toc}{section}{IxTLB — Insert TLB Entry}

\begin{iPageDescEntry}{Syntax}
    \texttt{IITLB RegR, RegB, RegX} \\
    \texttt{IDTLB RegR, RegB, RegX}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{IITLB} instruction uses the following instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0, 1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePTLB, RegR, 0, RegB, 0, RegX, 0}
    \end{flushleft}
    
    The \texttt{IDTLB} instruction uses the following instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0, 1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpSYS, \OpCodePTLB, RegR, 1, RegB, 0, RegX, 0}
    \end{flushleft}

\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{ITLB} instruction inserts a new entry into the translation lookaside buffer (TLB). The effective address is stored in \texttt{RegB} and additional control information is provided by \texttt{RegX}. This is a privileged instruction.
\end{iPageDescEntry}

\begin{iPageDescEntry}{TLB Arguments}
	\texttt{RegB}  and \texttt{RegA} contain the arguments passed to the TLB subsystem. \texttt{RegB}  contains the virtual address. The address is always o a page boundary and in the case of address ranges marks the starting page.
	
	\begin{flushleft}
	\drawInstrFormatSixtyFour
        {0, 3, 13,16}
        {0, VPN, 0}
    \end{flushleft}
    
    \texttt{RegA} contains the flags and the physical address of the address range. 
    
    \begin{flushleft}
	\drawInstrFormatSixtyFour
        {0, 2, 5, 6, 7, 13, 16}
        {Flags, 0, Acc, Size, PPN, 0}
    \end{flushleft}
 
	
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{RegR <- insertIntoTlb( RegB, RegA );}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    Privilege Violation Trap.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    None.  
\end{iPageDescEntry}
