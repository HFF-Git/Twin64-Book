\pagebreak
\section*{MTCR — Move To Control Register}
\addcontentsline{toc}{section}{MTCR — Move To Control Register}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{MTCR CReg, RegR}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{MTCR} instruction uses the control-register instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,16}
    {\OpCodeGrpSYS, \OpCodeMTCR, RegR, 1, CReg, 0}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{MTCR} instruction copies the contents of general register \texttt{RegR} to control register \texttt{CReg}.  
Some control registers are privileged. Attempting to write them in user mode results in a privilege exception.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
CReg <- RegR;    
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
Privilege operation trap for some control registers.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None
\end{iPageDescEntry}
