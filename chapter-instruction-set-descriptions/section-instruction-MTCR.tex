\pagebreak
\section*{MTCR — Move To Control Register}
\addcontentsline{toc}{section}{MTCR — Move To Control Register}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{MTCR CRegB, RegR}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{MTCR} instruction uses the control-register instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,16}
    {\OpCodeGrpSYS, \OpCodeMTCR, RegR, 1, CReg, 0}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{MTCR} instruction copies the contents of general register \texttt{RegR} to control register \texttt{CRegB}.  
Some control registers are privileged; attempting to write them in user mode may cause a privilege exception.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
CRegN <- RegR;    
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
Privilege exception if a privileged control register is written in user mode.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None
\end{iPageDescEntry}
