\pagebreak
\section*{MBR - Move and Branch}
\addcontentsline{toc}{section}{MBR - Move and Branch}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}

	\texttt{MBR.<cond> RegR, RegB, target}
    
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}

The MBR instruction uses the following format:
	
\begin{flushleft}
\drawInstrFormat
  	{0,1,3,5,6.5,8.5,16}
  	{\OpCodeGrpBR,\OpCodeMBR,RegR, cond, RegB, ofs }
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}

The \texttt{MBR} instruction copies the contents of \texttt{RegB} into \texttt{RegR}.  
If the condition specified in the \texttt{cond} field evaluates true for the new value in \texttt{RegR}, execution continues at the current instruction address plus the sign-extended offset. Otherwise, execution resumes with the next instruction.

\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Conditions}

The \texttt{cond} field encodes the branch condition as follows:

\begin{center}
\begin{tabular}{|c|c|l|}
\hline
\textbf{Value} & \textbf{Mnemonic} & \textbf{Branch if} \\ \hline
0 & \texttt{EQ} & \texttt{RegR == 0} \\ \hline
1 & \texttt{LT} & \texttt{RegR < 0}  \\ \hline
2 & \texttt{NE} & \texttt{RegR != 0} \\ \hline
3 & \texttt{LE} & \texttt{RegR <= 0} \\ \hline
4 & \texttt{EV} & \texttt{RegR.[0] == 0} \\ \hline
5 & \texttt{OD} & \texttt{RegR.[0] != 0} \\ \hline
6 & \texttt{rv} & reserved \\ \hline
7 & \texttt{rv} & reserved \\ \hline
\end{tabular}
\end{center}

\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- RegB;
if ( cond( RegR )) PSW.[IA] <- addOfs32( PSW.[IA], signExtend( ofs ) << 2 );
else               PSW.[IA] <- psw.[IA] + 4;
\end{lstlisting}
\end{iPageDescOpEntry}

	None.
\end{iPageDescEntry}
