\section*{LDIL - Load Immediate left}
\addcontentsline{toc}{section}{LDIL - Load Immediate left}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{LDIL[.L/S/U] RegR, val}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{LDIL} instruction uses the immediate-20 instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6,16}
    {\OpCodeGrpALU,\OpCodeLDIL, RegR, opt, val}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{LDIL} instruction loads an immediate value into the general register \texttt{RegR}.  
The optional suffix specifies how the immediate is positioned in the register:  

??? unclear text ... 

\begin{itemize}
    \item \texttt{.L} — Load into lower 20 bits and shift left by 12.  
    \item \texttt{.S} — Load into middle 20 bits and shift left by 32.  
    \item \texttt{.U} — Load into upper 12 bits and shift left by 52.
\end{itemize}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- ((val & 0xFFFFF) << 12)   // .L option
RegR <- ((val & 0xFFFFF) << 32)   // .S option
RegR <- ((val & 0xFFF)   << 52)   // .U option
\end{lstlisting}
\end{iPageDescOpEntry}
