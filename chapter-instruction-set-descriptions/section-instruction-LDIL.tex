\pagebreak
\section*{LDIL — Load Immediate Left}
\addcontentsline{toc}{section}{LDIL — Load Immediate Left}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{LDIL[.L/.S/.U] RegR, val}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{LDIL} instruction uses the immediate-20 instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6,16}
    {\OpCodeGrpALU,\OpCodeLDIL, RegR, mode, val}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{LDIL} instruction deposits an immediate value into the general register \texttt{RegR}.  
The optional suffix selects how the immediate field is positioned within the register.  
If no suffix is given, the default is \texttt{.L} mode (mode value = 1).  

\begin{itemize}
    \item \texttt{.L} (mode = 1) — Deposits 20 bits of \texttt{val} into bits [31..12] of \texttt{RegR}.  
    \item \texttt{.S} (mode = 2) — Deposits 20 bits of \texttt{val} into bits [51..32] of \texttt{RegR}.  
    \item \texttt{.U} (mode = 3) — Deposits the lower 12 bits of \texttt{val} into bits [63..52] of \texttt{RegR}.
\end{itemize}

A mode value of zero is reserved and must not be used.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- ((val & 0xFFFFF) << 12)   // .L mode (1)
RegR <- ((val & 0xFFFFF) << 32)   // .S mode (2)
RegR <- ((val & 0xFFF)   << 52)   // .U mode (3)
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
None.
\end{iPageDescEntry}
