\pagebreak
\section*{ST — Store Register}
\addcontentsline{toc}{section}{ST — Store Register}

\begin{iPageDescEntry}{Syntax}
    \texttt{ST[.D/W/H/B] RegR, SignedImmed13( RegB )} \\
    \texttt{ST[.D/W/H/B] RegR, RegX( RegB )}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{ST} instruction uses the indexed and register-indexed instruction formats. \\

    \begin{flushleft}
    \drawInstrFormat
        {0,1,3,5,6.5,8.5,9.5,16}
        {\OpCodeGrpMEM, \OpCodeST, RegR, 0, RegB, dw, S-IMM-13}
    \end{flushleft}

    \begin{flushleft}
    \drawInstrFormat
        {0,1,3,5,6.5,8.5,9.5,11.5,16}
        {\OpCodeGrpMEM, \OpCodeST, RegR, 1, RegB, dw, RegX, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{ST} instruction stores a value from a general-purpose register into memory. 
    
    In the immediate-offset format, the signed immediate offset is shifted right by the \texttt{dw} field of the instruction and added to \texttt{RegB} to form the effective address. In the register-indexed format, \texttt{RegX} is added to \texttt{RegB} to form the effective address.
    
    The store address must be aligned to the data width specified. The value stored to memory is stored  as is in the data width specified.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
memStoreImmIndexed( instr, RegR );
memStoreRegIndexed( instr, RegR );
\end{lstlisting}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    - Data Alignment Trap \\
    - Memory Access Violation Trap \\
    - Memory Protection Violation Trap \\
    - Data TLB Miss Trap
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
   None.
\end{iPageDescEntry}
