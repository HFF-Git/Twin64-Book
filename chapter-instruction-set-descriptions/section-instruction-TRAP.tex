\pagebreak
\section*{TRAP — Trap Instruction}
\addcontentsline{toc}{section}{TRAP — Trap Instruction}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{TRAP tid, regB, regA}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{TRAP} instruction uses the standard indexed instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,9.5,11.5,16}
    {\OpCodeGrpSYS,\OpCodeTRAP, 0, tid1, RegB, tid2, RegA, 0}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{TRAP} instruction generates a synchronous trap to handle exceptional conditions.  The \texttt{tid1} and \texttt{tid2} are concatenated to form the trap Id. The instruction may be used to invoke operating system routines or exception handlers.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
raiseTrap( id1, RegB, RegA );
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
The trap raised.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None
\end{iPageDescEntry}
