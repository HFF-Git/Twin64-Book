\section*{FCA - flush from data cache}
\addcontentsline{toc}{section}{FCA - flush from cache}

\begin{iPageDescEntry}{Syntax}
    \texttt{FCA RegR, RegB}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Format}
    The \texttt{FCA} instruction uses the standard indexed instruction format. \\

    \begin{flushleft}
    \drawInstrFormat
        {0,1,3,5,6.5,8.5,16}
        {\OpCodeGrpSYS, \OpCodeFCA, RegR, 1, RegB, 0}
    \end{flushleft}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Description}
    The \texttt{FCA} instruction flushes a cache line, selected by the effective address
    in \texttt{RegB}. This is a privileged instruction.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Operation}
    \texttt{RegR <- flushFromCache( RegB );}
\end{iPageDescEntry}

\begin{iPageDescEntry}{Exceptions}
    Privilege violation trap.
\end{iPageDescEntry}

\begin{iPageDescEntry}{Notes}
    We may enhance this instruction with another register to for the target address by adding RegB with RegX, so that we can better support flushing cache ranges.
\end{iPageDescEntry}
