\pagebreak
\section*{BB — Branch on Bit}
\addcontentsline{toc}{section}{BB — Branch on Bit}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax} 
  \texttt{BB[.T/F] ofs, pos} \\
  \texttt{BB[.T/F] ofs, SAR}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}

The \texttt{BB} instruction uses the following instruction format: \\

\begin{flushleft}
  \drawInstrFormat
    {0,1,3,5,5.5,6,6.5,9.5,16}
    {\OpCodeGrpBR,\OpCodeBB,RegR,0,A,V,pos,ofs}
\end{flushleft}

\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}

The \texttt{BB} instruction tests a bit in register \texttt{RegR}.  
The bit position can be specified directly by the \texttt{pos} field,  
or indirectly through the Shift Amount Register (\texttt{SAR}) if the \texttt{A} option is set.  

If the \texttt{V} bit is set, the test succeeds when the selected bit is \texttt{1}.  
If the \texttt{V} bit is clear, the test succeeds when the selected bit is \texttt{0}.  

When the condition is satisfied, execution branches to the target address,  
computed by sign-extending the offset, shifting it left by two bits,  
and adding it to the current instruction address. If the condition is not satisfied, execution continues with the next sequential instruction.  

\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
if ( instr.[A] )  pos <- SAR;
else              pos <- instr.[pos];

bVal <- testBit( RegR, pos );
cond <- ( instr.[V] ? ( bVal == 1 ) : ( bVal == 0 ));

if ( cond ) {

    tmpOfs <- signExtend( ofs ) << 2;
    PSR.[IA-OFS] <- addOfs32( PSR.[IA-OFS], tmpOfs );
}
else PSR.[IA-OFS] <- addOfs32( PSR.[IA-OFS], 4 );
\end{lstlisting}
\end{iPageDescOpEntry}
