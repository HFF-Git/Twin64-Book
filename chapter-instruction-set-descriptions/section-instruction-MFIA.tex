\pagebreak
\section*{MFIA - Move from Instruction Address}
\addcontentsline{toc}{section}{MFIA - Move from Instruction Address}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{MFIA RegR}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{MFIA} instruction uses the register instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,16}
    {\OpCodeGrpSYS, \OpCodeMFIA, RegR, 2, 0}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{MFIA} instruction copies the current instruction address to register \texttt{RegR}.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- PSW.[IA];
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
None.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None.
\end{iPageDescEntry}
