\pagebreak
\section*{MFIA - Move from Instruction Address}
\addcontentsline{toc}{section}{MFIA - Move from Instruction Address}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{MFIA [.R/L/A]  RegR}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{MFIA} instruction uses the register instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,5.5, 6.5,16}
    {\OpCodeGrpSYS, \OpCodeMFIA, RegR, 1, mode, 0}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{MFIA} instruction copies the current instruction address to register \texttt{RegR}.The mode field allows to select a portion of the instruction address. 

\begin{itemize}
 	\setlength{\itemsep}{0.5em}
    \setlength{\parskip}{0pt}
	\item \texttt{.A} (mode = 0) — copies IA to \texttt{RegR}. This is the default operation.
    \item \texttt{.L} (mode = 1) — Deposits 20 bits of \texttt{IA.[31..12]} into bits [31..12] of \texttt{RegR}. \texttt{RegR} is cleared before.
    \item \texttt{.R} (mode = 2) — Deposits 20 bits of \texttt{IA.[51..32]} into bits [51..32] of \texttt{RegR}. \texttt{RegR} is cleared before. 
\end{itemize}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- PSW.[IA]                          // .A mode (0)
RegR <- (( PSW.[IA] & 0xFFFFF ) << 12 )   // .L mode (1)
RegR <- (( PSW.[IA] & 0xFFFFF ) << 32 )   // .R mode (2)
\end{lstlisting}
\end{iPageDescOpEntry}
