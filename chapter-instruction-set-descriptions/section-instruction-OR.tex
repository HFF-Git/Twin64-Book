\section*{OR Boolean Operation}
\addcontentsline{toc}{section}{OR Boolean Operation}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
  \texttt{OR[.C/N] RegR, RegB, RegA} \\
  \texttt{OR[.C/N] RegR, RegB, val}
  
  \texttt{OR[.C/N/D/W/H/B] RegR, ofs( RegB )} \\
  \texttt{OR[.C/N/D/W/H/B] RegR, RegX ( RegB )}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
  The OR instruction uses the following instruction formats: \\
	
  \begin{flushleft}	
  \drawInstrFormat
  	{0,1,3,5,5.5,6,6.5,8.5,9.5,11.5,16}
  	{\OpCodeGrpALU,\OpCodeOR, RegR, N, C, 0, RegB, 0, RegA, 0}
	
  \drawInstrFormat
  	{0,1,3,5,5.5,6,6.5,8.5,16}
  	{\OpCodeGrpALU,\OpCodeOR, RegR, N, C, 1, RegB, val}

  \drawInstrFormat
  	{0,1,3,5,5.5,6,6.5,8.5,9.5,16}
  	{\OpCodeGrpMEM,\OpCodeOR, RegR, N, C, 0, RegB, dw, ofs}
 
  \drawInstrFormat
  	{0,1,3,5,5.5,6,6.5,8.5,9.5,11.5,16}
  	{\OpCodeGrpMEM,\OpCodeOR, RegR, N, C, 1, RegB, dw, RegX, 0}
  \end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
  The \texttt{OR} instruction performs a bitwise OR operation on two 64-bit operands and stores the result in the target register \texttt{RegR}. The \texttt{AND} instruction performs a bitwise OR operation on two 64-bit operands and stores the result in the target register \texttt{RegR}. The second operand, \texttt{RegB} or the loaded content, can be negated by setting the "C" bit, and the result can be negated by setting the "N" bit. This instruction supports the immediate, register, and both memory-reference formats. The boolean operation itself does not trigger any traps, but the indexed formats may cause a memory-reference related trap.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]

tmp <- RegA;                            (S9)
tmp <- immOperand( instr );             (S15)
tmp <- memOperandImmIndexed( instr );   (S13)
tmp <- memOpwerandRegIndexed( instr );  (S9 )

if ( instr.[C] ) tmp <- not( tmp );
RegR <- RegB | tmp;
if ( instr.[N] ) RegR <- not( RegR );
\end{lstlisting}
\end{iPageDescOpEntry}
