\section*{LDO — Load Offset}
\addcontentsline{toc}{section}{LDO — Load Offset}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{LDO RegR, ofs(RegB)}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{LDO} instruction uses the immediate-15 instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,16}
    {0,\OpCodeLDO,RegR,0,RegB,ofs}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{LDO} instruction loads a value from memory using an offset addressing mode.  
The effective address is calculated by adding the signed immediate offset \texttt{ofs} to the contents of \texttt{RegB}.  
The loaded value is then stored in \texttt{RegR}.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- addOfs32(RegB, ofs);
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
None.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None
\end{iPageDescEntry}
