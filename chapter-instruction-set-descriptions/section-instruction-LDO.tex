\section*{LDO — Load Offset}
\addcontentsline{toc}{section}{LDO — Load Offset}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{LDO [.B/H/W/D] RegR, ofs(RegB)}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{LDO} instruction uses the immediate-15 instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,9.5,16}
    {0,\OpCodeLDO,RegR,0,RegB, dw, ofs}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{LDO} instruction loads a value from memory using an offset addressing mode. The \texttt{ofs} value is sign extended and shifted by the \texttt{dw} field. The result is calculated by adding this value to the contents of \texttt{RegB}.  
The loaded value is then stored in \texttt{RegR}.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- addOfs32(RegB, signExtend( ofs ) << dw );
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
None.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None
\end{iPageDescEntry}
