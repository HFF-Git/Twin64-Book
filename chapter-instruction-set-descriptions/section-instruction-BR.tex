\section*{BR — Branch Register}
\addcontentsline{toc}{section}{BR — Branch Register}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{BR RegB [, RegR ]}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Format}
The \texttt{BR} instruction uses the immediate-15 instruction format. \\

\begin{flushleft}
\drawInstrFormat
    {0,1,3,5,6.5,8.5,16} % adjust positions as needed
    {\OpCodeGrpBR, \OpCodeBR, RegR, 0, RegB, 0}
\end{flushleft}
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}
The \texttt{BR} instruction performs an unconditional branch to a target address computed relative to the current instruction address segment (\texttt{IA}). The target address is formed by taking the contents of \texttt{RegB}, shifting it left by two bits, and adding it to the \texttt{IA} offset portion. The return link is stored in \texttt{RegR}.
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Operation}
\begin{lstlisting}[style=iPageOpStyle]
RegR <- addOfs32( PSW.[IA], 4 );
PSW.[IA] <- addOfs32( PSW.[IA], RegB << 2 );
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Exceptions}
Taken Branch Trap\\
Instruction TLB Miss Trap
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None.
\end{iPageDescEntry}
