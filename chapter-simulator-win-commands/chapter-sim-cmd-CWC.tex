\section*{CWC - Clear command window}
\addcontentsline{toc}{section}{CWC -  Clear command window}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Syntax}
\texttt{CWC } \\
\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Description}

The \texttt{CWC} command clears the command window. All other widows are not affected.

\end{iPageDescEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescOpEntry}{Example}
\begin{lstlisting}[style=iPageOpStyle]
cwc                # clears the command window.
\end{lstlisting}
\end{iPageDescOpEntry}

%--------------------------------------------------------------------------------
\begin{iPageDescEntry}{Notes}
None.
\end{iPageDescEntry}
